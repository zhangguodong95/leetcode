### 常用排序（归并 ，快速，冒泡，插入，选择）

```java

//归并排序
public void mergeSort(int[] arr, int left, int right) {
    if (left >= right) return;
    int mid = (left + right) >> 1;
    mergeSort(arr,left,mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}
private void merge(int[] arr, int left, int mid, int right) {
    //申请一个新的数组
    int[] res = new int[right - left + 1];
    int i = left, j = mid + 1, k = 0;
    while (i <= mid && j <= right) res[k++] = arr[i] < arr[j] ? arr[i++] : arr[j++];
    while (i <= mid) res[k++] = arr[i++];
    while (j <= right) res[k++] = arr[j++];
    //拷贝数组
    System.arraycopy(res,0,arr,left,right - left + 1);
}

//快速排序
public void quickSort(int[] arr, int left, int right) {
    if(left >= right) return;
    int position = partition(arr, left, right);
    quickSort(arr,left,position - 1);
    quickSort(arr, position + 1, right);
}
private int partition(int[] arr, int left, int right) {
    int pivot = arr[right];
    int current = left;
    for (int i = left; i < right; i++) {
        if (arr[i] < pivot) {
            int temp = arr[i];
            arr[i] = arr[current];
            arr[current] = temp;
            current++;
        }
    }
    int temp = arr[right]; arr[right] = arr[current]; arr[current] = temp;
    return current;
}

// 冒泡排序
public void bubbleSort(int[] arr, int n) { //n为数组的大小
    if (n == 0 || arr == null) return;
    for (int i = 1; i < n; i++) { //要n - 1次循环
        for (int j = 0; j < n - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

//插入排序
public void insertSort(int[] arr, int n) {
    for (int i = 1; i < n; i++) {
        int current = arr[i];
        for (int j = i - 1; j >= 0; j--) {//从后往前比较
            if (current > arr[j]) {
                //j + 1的位置即是要插入的位置
                arr[j + 1] = current;
                break;
            } else {
                arr[j + 1] = arr[j];
            }
        }
    }
}

//选择排序 每次选出最小的
public void selectSort(int[] arr, int n) {
    for (int i = 0; i < n - 1; i++) {
        //选出最小值
        int min = arr[i];
        int pos = i;
        for (int j = i; j < n; j++) {
            if (arr[j] < min) {
                min = arr[j];
                pos = j;
            }
        }
        int temp = arr[pos];
        arr[pos] = arr[i];
        arr[i] = temp;
    }
}
```
